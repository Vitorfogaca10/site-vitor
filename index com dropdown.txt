import { NavLink } from "react-router-dom";

import { RiArrowUpWideFill } from "react-icons/ri";

function Home() {
    const [openLogin, setOpenLogin] = useState(false);
    const [openCadastro, setOpenCadastro] = useState(false);

    return (
        <header className="navbar">
            <div className="container">
                <div className="logo">BRAPY</div>
                <nav className="link-btt">
                    <ul className="links">

                    <ul className="nav-login">

                            {openLogin && (
                                <div className="dropdown">
                                    <NavLink to="/login-cliente">Entrar como cliente</NavLink>
                                    <NavLink to="/login-freteiro">Entrar como freteiro</NavLink>
                                </div>
                            )}
                        </li>
                    </ul>
                    <ul className="nav-cadastro ">
                        {/* Botão de login com dropdown */}
                        <li style={{ position: "relative" }}>
                            <button onClick={() => { setOpenCadastro(!openCadastro); setOpenLogin(false); }} className="menu-btn">Cadastre-se{" "}<RiArrowUpWideFill style={{ marginLeft: "5px", transform: openCadastro ? "rotate(180deg)" : "rotate(0deg)", transition: "transform 0.3s", }} /></button>
                            {openCadastro && (
                                <div className="dropdown">

                                </div>
                            )}
                        </li>
                    </ul>
                </nav>
            </div>
            <main className="main">
                <div className="conteudo">
                    <h1>Conectamos quem precisa importar <br/>com quem já está vindo do Paraguai</h1>
                    <p>Importe com confiança e pague menos. Transporte com segurança e ganhe dinheiro extra.</p>

            </main>
        </header>
